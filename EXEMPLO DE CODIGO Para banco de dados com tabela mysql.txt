CREATE TABLE TB_CATEGORIA (
    I_COD_CATEGORIA SERIAL  AUTO_INCREMENT,
    S_NM_CATEGORIA VARCHAR(255),
    PRIMARY KEY(I_COD_CATEGORIA)
);

CREATE TABLE TB_PRODUTO (
    I_COD_PRODUTO SERIAL AUTO_INCREMENT,
    I_COD_CATEGORIA INT,
    S_NM_PRODUTO VARCHAR(255),
    D_VLR_PRODUTO DECIMAL(10, 2),
    PRIMARY KEY(I_COD_PRODUTO),
    FOREIGN KEY (I_COD_CATEGORIA) REFERENCES TB_CATEGORIA(I_COD_CATEGORIA)
);


CREATE TABLE TB_PEDIDOS (
    I_COD_PEDIDO SERIAL AUTO_INCREMENT,
    I_COD_CLIENTE INT,
    DT_DATA_PEDIDO DATE,
    D_TOT_PEDIDO DECIMAL(10, 2),
    PRIMARY KEY(I_COD_PEDIDO),
    FOREIGN KEY (I_COD_CLIENTE) REFERENCES TB_CLIENTE(I_COD_CLIENTE)
);


CREATE TABLE TB_ITENS (
    I_COD_ITENS SERIAL AUTO_INCREMENT,
    I_COD_PEDIDO INT,
    I_COD_PRODUTO INT,
    I_QTDE_ITENS INT,
    D_VLRUNIT_ITENS DECIMAL(10, 2),
    PRIMARY KEY(I_COD_ITENS),
    FOREIGN KEY (I_COD_PEDIDO) REFERENCES TB_PEDIDO(I_COD_PEDIDO)
);

CREATE TABLE TB_CLIENTE (
    I_COD_CLIENTE SERIAL AUTO_INCREMENT,
    S_NM_CLIENTE VARCHAR(255),
    S_MAIL_CLIENTE VARCHAR(255),
    S_CEL_CLIENTE VARCHAR(20),
    S_CNPJ_CLIENTE VARCHAR(20),
    PRIMARY KEY(I_COD_CLIENTE)
);

CREATE TABLE TB_USUARIO (
 	I_COD_USUARIO SERIAL AUTO_INCREMENT,
	S_NOME_USUARIO VARCHAR(255),
	S_PASS_USUARIO VARCHAR(255),
	S_MAIL_USUARIO VARCHAR(255),
	I_NIVEL_USUARIO INT,
	PRIMARY KEY(I_COD_USUARIO)
);
